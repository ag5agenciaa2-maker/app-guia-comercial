    function App() {
        const [currentPage, setCurrentPage] = useState('home');
        const [isChatOpen, setIsChatOpen] = useState(false);
        const [chatMessages, setChatMessages] = useState([]);
        const [chatInput, setChatInput] = useState('');
        const [searchTerm, setSearchTerm] = useState('');
        const [searchResults, setSearchResults] = useState([]);
        const [showCompanyDetails, setShowCompanyDetails] = useState(false);
        const [motivationalMessage, setMotivationalMessage] = useState(motivationalQuotes[0]);
        const [modalInfo, setModalInfo] = useState({ isOpen: false, title: '', message: '' });

        const showAlertModal = (title, message) => {
            setModalInfo({ isOpen: true, title, message });
        };
        const closeModal = () => {
            setModalInfo({ isOpen: false, title: '', message: '' });
        };

        const appBgColor = '#1a1a1a';

        useEffect(() => {
            const intervalId = setInterval(() => {
                const randomIndex = Math.floor(Math.random() * motivationalQuotes.length);
                setMotivationalMessage(motivationalQuotes[randomIndex]);
            }, 12 * 60 * 60 * 1000);

            return () => clearInterval(intervalId);
        }, []);

        const handleAIMessage = async (message) => {
            setChatMessages((prev) => [...prev, { sender: 'user', text: message }]);
            setChatInput('');

            let aiResponse = "Desculpe, não entendi sua pergunta. Posso ajudar com informações sobre os serviços da AG5, pesquisar empresas, ou sobre as funcionalidades do app.";

            if (message.toLowerCase().includes('olá') || message.toLowerCase().includes('oi')) {
                aiResponse = "Olá! 👋 Como posso te ajudar hoje? Explore as funcionalidades do app ou me pergunte sobre os serviços da AG5.";
            } else if (message.toLowerCase().includes('serviços')) {
                aiResponse = "A AG5 oferece: Atualização e Gerenciamento de Perfis no Google Meu Negócio, Foto 360°, Tour Virtual, Gerenciamento de Instagram, Tráfego Pago, e produtos como Placa Acrílica de Avaliações e Sites. 🚀";
            } else if (message.toLowerCase().includes('equipe')) {
                aiResponse = "Nossa equipe inclui: Mauricio Areia (Diretor Operacional), Fagner Galvão (Diretor de Vendas), Demilson (Comercial), Helem Brito e Mylena Martins (Atualização de Perfil), Sara Areia (Gerenciamento de Perfil), Jullyane (Gerenciamento de Instagram), e parceiros para Criação de Site e Produtos. ✔️";
            } else if (message.toLowerCase().includes('pesquisar empresa')) {
                aiResponse = "Para pesquisar uma empresa, digite o nome, categoria, produto ou serviço na barra de pesquisa da tela inicial. Os resultados aparecerão abaixo. Ao clicar em um resultado, você será direcionado para uma pesquisa no Google sobre a empresa. 🔍";
            } else if (message.toLowerCase().includes('contato') || message.toLowerCase().includes('whatsapp')) {
                aiResponse = "Para falar diretamente com a equipe, use o botão 'Ir para Grupo' na tela inicial, ou acesse os links no rodapé da página. 📱";
            } else if (message.toLowerCase().includes('notas e metas') || message.toLowerCase().includes('previsão') || message.toLowerCase().includes('comissão')) {
                aiResponse = "A página de 'Notas e Metas' está atualmente em construção. Em breve teremos novidades sobre essa funcionalidade! 😉";
            } else if (message.toLowerCase().includes('obrigado') || message.toLowerCase().includes('valeu')) {
                aiResponse = "De nada! Estou aqui para ajudar. 😊";
            }

            setTimeout(() => {
                setChatMessages((prev) => [...prev, { sender: 'ai', text: aiResponse }]);
            }, 500);
        };

        const handleSearch = (e) => {
            const term = e.target.value.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
            setSearchTerm(e.target.value);
            if (term.trim().length > 0) {
                const filtered = companiesData.filter(company => {
                    const companyNameNormalized = company.name.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
                    const companyCategoryNormalized = company.category.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").replace(/[^\p{L}\p{N}\s]/gu, '').trim();
                    const companyKeywordsNormalized = company.keywords.map(kw => kw.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, ""));

                    // Correção: Garante que a pesquisa seja mais específica para a categoria.
                    // Se o termo de busca estiver nas palavras-chave, verifica se a categoria da empresa também
                    // é relevante ou se o termo é um nome/categoria direta.
                    return companyNameNormalized.includes(term) ||
                           companyCategoryNormalized.includes(term) ||
                           companyKeywordsNormalized.some(kw => kw.includes(term) || term.includes(kw));
                });
                setSearchResults(filtered);
                setShowCompanyDetails(true);
            } else {
                setSearchResults([]);
                setShowCompanyDetails(false);
            }
        };

        const renderPage = () => {
            const commonProps = { setCurrentPage, appBgColor, showAlertModal };
            switch (currentPage) {
                case 'home':
                    return React.createElement(HomePage, {
                        ...commonProps,
                        motivationalMessage,
                        searchTerm,
                        handleSearch,
                        showCompanyDetails,
                        searchResults,
                        setShowCompanyDetails,
                        isChatOpen,
                        setIsChatOpen,
                        chatMessages,
                        chatInput,
                        handleAIMessage,
                        setChatInput
                    });
                case 'client-results':
                    return React.createElement(ClientResultsPage, commonProps);
                case 'sales-insights':
                    return React.createElement(SalesInsightsPage, commonProps);
                default:
                    return React.createElement(HomePage, {
                        ...commonProps,
                        motivationalMessage,
                        searchTerm,
                        handleSearch,
                        showCompanyDetails,
                        searchResults,
                        setShowCompanyDetails,
                        isChatOpen,
                        setIsChatOpen,
                        chatMessages,
                        chatInput,
                        handleAIMessage,
                        setChatInput
                    });
            }
        };

        return (
            React.createElement("div", { className: "font-sans antialiased" },
                renderPage(),
                React.createElement(Modal, { isOpen: modalInfo.isOpen, onClose: closeModal, title: modalInfo.title }, modalInfo.message)
            )
        );
    }

    ReactDOM.render(React.createElement(App), document.getElementById('root'));
    </script>
